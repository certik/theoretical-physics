>>> var("n")
n

>>> A_2 = S(197)/144 + zeta(2)/2 + 3*zeta(3)/4 - 3*zeta(2) * log(2)
>>> A_3 = 83*pi**2*zeta(3)/72 - 215*zeta(5)/24 + 100*(a4 + log(2)**4/24 - \
...         pi**2*log(2)**2/24)/3 - \

...         17101 * pi**2 / 810 + S(28259)/5184
